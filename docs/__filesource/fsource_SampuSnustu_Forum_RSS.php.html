  31.    /**
  32.     * Erzeugt den RSS-XML-Code des in der URL spezifizierten FObject-Objektes
  33.     * und gibt ihn zurück
  34.     * 
  35.     * @return string erzeugter RSS-XML-Code wenn erfolgreich, ansonsten ''
  36.     */
  37.    public static function createByRequest() {
  38.       if (isset($_REQUEST["fid"]) && is_numeric($_REQUEST["fid"])) {
  39.          return self::createFObjectRSS(FSystem::getForumById($_REQUEST["fid"]));
  40.       } else if (isset($_REQUEST["tid"]) && is_numeric($_REQUEST["tid"])) {
  41.          return self::createFObjectRSS(FSystem::getThreadById($_REQUEST["tid"]));
  42.       }
  43.       return '';
  44.    }
  45.  
  46.    /**
  47.     * Erzeugt den RSS-XML-Code des übergebenen Thread-Objektes und gibt ihn zurück
  48.     * 
  49.     * @param Thread $fobject Thread-Objekt
  50.     * @return string erzeugter RSS-XML-Code wenn erfolgreich, ansonsten ''
  51.     */
  52.    public static function createThreadRSS($thread) {
  53.       if ($thread) {
  54.          return self::createFObjectRSS(get_class($thread) == "Thread" ? $thread : FSystem::getThreadById($thread));
  55.       }
  56.       return '';
  57.    }
  58.  
  59.    /**
  60.     * Erzeugt den RSS-XML-Code des übergebenen FObject-Objekt und gibt ihn zurück
  61.     * 
  62.     * @param FObject $fobject FObject-Objektes
  63.     * @param boolean $setheader wenn true, wird der Header auf
  64.     *  "Content-Type: application/rss+xml" gesetzt
  65.     * @return string erzeugter RSS-XML-Code wenn erfolgreich, ansonsten ''
  66.     */
  67.    public static function createFObjectRSS(FObject $fobject, $setheader = true) {
  68.       if ($fobject && is_a($fobject, 'Thread')) {
  69.          if ($setheader) {
  70.             header("Content-Type: application/rss+xml");
  71.          }
  72.          $entries = $fobject->getLastEntries();
  73.          $html = '
  74.           <?xml version="1.0" encoding="ISO-8859-1"?>
  75.             <rss version="2.0">
  76.             <channel>
  77.                 <title>' . $fobject->getTitle() . '</title>
  78.                 <link>' . FSystem::URL . 'forum.php?' . (get_class($fobject) == 'Forum' ? 'fid' : 'tid') . '=' . $fobject->getID() . '</link>
  79.                 <description>' . $fobject->getDescription() . '</description>
  80.                 <language>de-de</language>
  81.                 <pubDate>' . $entries ? date("r", $entries[count($entries) - 1]->getMTime()) : date("r") . '</pubDate>';
  82.          if ($entries) {
  83.             foreach ($entries as $entry) {
  84.                $html .= '  
  85.                     <item>
  86.                         <title>' . $entry->getThread()->getTitle() . ' | ' . $entry->getUser()->getName() . '</title>
  87.                         <link>' . FSystem::URL . 'forum.php?method=show&eid=' . $entry->getID() . '</link>
  88.                         <description><![CDATA[' . $entry->getContent() . ']]></description>
  89.                         <pubDate>' . date("r", $entry->getCTime()) . '</pubDate>
  90.                     </item>';
  91.             }
  92.          }
  93.          return $html . '
  94.                 </channel>
  95.             </rss>';
  96.       }
  97.       return '';
  98.    }

  111.                 </a>';
  112.       }
  113.       return '';
  114.    }
