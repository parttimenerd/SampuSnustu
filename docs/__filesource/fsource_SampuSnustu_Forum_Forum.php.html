  133.    public function getEntries($begin, $length) {
  134.       $entryarr = array();
  135.       FDB::connect();
  136.       $result = FDB::$db->query('SELECT * FROM forum WHERE parent=' . $this->ID . ' ORDER BY title LIMIT ' . $begin . ', ' . $length);

  138.       while ($entryarr[] = Forum::createFromFDBResult($result)) {
  139.          $anz++;
  140.       }
  141.       $length -= $anz;
  142.       $begin -= $anz;
  143.       if ($begin < 0) {
  144.          $begin = 0;
  145.       }
  146.       if ($length > 0) {
  147.          $result = FDB::$db->query('SELECT * FROM thread WHERE parent=' . $this->ID . ' ORDER BY mtime DESC LIMIT ' . $begin . ', ' . $length);
  148.          while ($entryarr[] = Thread::createFromFDBResult($result));
  149.       }
  150.       return $entryarr;
  151.    }
