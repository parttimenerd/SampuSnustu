  144.    private function createAudioFile() {
  145.       $wavs = array();
  146.       $chars = str_split($this->text, 1);
  147.       foreach ($chars as $char) {
  148.          $wavs[] = __DIR__ . '/' . self::AUDIO_PART_FILE_DIR . $char . '.wav';
  149.       }
  150.       $data = $this->joinwavs($wavs);
  151.       $file = fopen($this->path . '/' . self::AUDIO_FILENAME, 'w');
  152.       fwrite($file, $data);
  153.    }

  155.    /**
  156.     * Verbindet mehrere Wav-Dateien zu einem Datenarray, via
  157.     * http://www.splitbrain.org/blog/2006-11/15-joining_wavs_with_php
  158.     * @param String[] $wavs Wav-Dateinamen
  159.     * @return array() Datenarray der zusammengefügten Dateien
  160.     * @copyright www.splitbrain.org/
  161.     * @author www.splitbrain.org/
  162.     */
  163.    private function joinwavs($wavs) {
  164.       $fields = join('/', array('H8ChunkID', 'VChunkSize', 'H8Format',
  165.           'H8Subchunk1ID', 'VSubchunk1Size',
  166.           'vAudioFormat', 'vNumChannels', 'VSampleRate',
  167.           'VByteRate', 'vBlockAlign', 'vBitsPerSample'));
  168.       $data = '';
  169.       $header = '';
  170.       foreach ($wavs as $wav) {
  171.          $fp = fopen($wav, 'rb');
  172.          $header = fread($fp, 36);
  173.          $info = unpack($fields, $header);
  174.          // read optional extra stuff
  175.          if ($info['Subchunk1Size'] > 16) {
  176.             $header .= fread($fp, ($info['Subchunk1Size'] - 16));
  177.          }
  178.          // read SubChunk2ID
  179.          $header .= fread($fp, 4);
  180.          // read Subchunk2Size
  181.          $size = unpack('vsize', fread($fp, 4));
  182.          $size = $size['size'];
  183.          // read data
  184.          $data .= fread($fp, $size * 10);
  185.       }
  186.       return $header . pack('V', strlen($data)) . $data;
  187.    }

  192.    private function createImageFile() {
  193.       $image = imagecreate($this->width, $this->height);
  194.       $chars = str_split($this->text, 1);
  195.       $w = $this->width - ($this->width / 4);
  196.       $x = ($this->height / 8);
  197.       $minfontsize = $this->height / 4;
  198.       $maxfontsize = $this->height / 1.7;
  199.       $blue = imagecolorallocate($image, 0, 0, 255);
  200.       imagefilledrectangle($image, 0, 0, $this->width, $this->height, imagecolorallocatealpha($image, 255, 255, 255, 127));
  201.       for ($i = 0; $i < count($chars); $i++) {
  202.          //imagechar($image, 10, $x, $y, $chars[$i], $blue);
  203.          $fontsize = rand($minfontsize, $maxfontsize);
  204.          imagettftext($image, $fontsize, rand(0, 45), $x, $this->height - ($fontsize / 2), $blue, __DIR__ . '/' . self::FONTFILE, $chars[$i]);
  205.          $x += $w / count($chars) - rand(-$minfontsize / 3, $minfontsize / 3); //$i + rand($this->width / count($chars) / 3, $this->width / count($chars) / 2);
  206.       }
  207.       for ($i = 0; $i < $this->height * $this->width / 1.8; $i++) {
  208.          imagesetpixel($image, rand(0, $this->width), rand(0, $this->height), $blue);
  209.       }
  210.       imagepng($image, $this->path . '/' . self::IMAGE_FILENAME);
  211.    }
